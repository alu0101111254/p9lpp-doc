def to_s
  resultado=""
  resultado+="Nombre: "+@name+ "\n"
  resultado+="alimentos: \n"
  @list.each do |var|
    resultado+=var.to_s
    resultado+="\n"
  end
  resultado+="pesos: \n"
  @grams.each do |var|
    resultado+=var.to_s
    resultado+="\n"
  end
  resultado+="porcentaje proteinas: "+ prot.to_s + " \n"
  resultado+="porcentaje lipidos: "+ lipid.to_s + " \n"
  resultado+="porcentaje carbohidratos: "+ carbs.to_s + " \n"
  resultado+="VCT: "+ vct.to_s + " \n"
end

def total_nutr
  tot_gram=0.0
  @list.each do |var|
    tot_gram+=var.proteins
    tot_gram+=var.lipids
    tot_gram+=var.carbos
  end
  return tot_gram
end
